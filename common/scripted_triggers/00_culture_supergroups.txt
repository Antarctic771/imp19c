### Culture super-group triggers
basic_western_group_trigger = {
	OR = {
		culture_group = english_group
		culture_group = angloamerican
		culture_group = spanish
		culture_group = portuguese_group
		culture_group = scandinavian
		culture_group = german
		culture_group = dutch_group
		culture_group = italian
		culture_group = french_group
		culture_group = north_celtic
	}
}

greater_slavic_group_trigger = {
	OR = {
		culture_group = east_slavic
		culture_group = west_slavic
		culture_group = south_slavic
	}
}

greater_hispanic_group_trigger = {
	OR = {
		culture_group = spanish
		culture_group = north_latino
		culture_group = south_latino
	}
}

greater_western_group_trigger = {
	OR = {
		basic_western_group_trigger = yes
		greater_slavic_group_trigger = yes
		culture_group = hellenic
		culture_group = north_latino
		culture_group = south_latino
	}
}

indo_aryan_group_trigger = {
	OR = {
		culture_group = central_aryan
		culture_group = southern_aryan
		culture_group = western_aryan
		culture_group = eastern_aryan
		culture_group = gurkani
	}
}

greater_indian_group_trigger = {
	OR = {
		indo_aryan_group_trigger = yes
		culture_group = dravidian
		culture_group = north_dravidian
	}
}
